\chapter[{TỔNG KẾT VÀ ĐỊNH HƯỚNG TƯƠNG LAI}]{TỔNG KẾT VÀ ĐỊNH HƯỚNG TƯƠNG LAI}

\section{Tổng kết}
Hệ thống quản lí tài chính cá nhân đơn giản được xây dựng với mục đích giúp người dùng dễ dàng ghi chép, quản lý, theo dõi và phân tích chi tiêu.\\
Với giao diện thân thiện và các chức năng trực quan, hệ thống hướng đến việc hỗ trợ người dùng kiểm soát chi tiêu hiệu quả và xây dựng thói quen tài chính lành mạnh. \\
\subsection{Điểm cộng của hệ thống}
\begin{itemize}
	\item Giao diện thân thiện người dùng phổ thông.
	\item Chức năng quản lý tài chính cơ bản đầy đủ, hiệu quả trong việc theo dõi chi tiêu.
	\item Tích hợp AI ChatBot giúp người dùng đưa ra quyết định tài chính tốt hơn.	
	\item Khả năng tùy chỉnh linh hoạt, người dùng có thể tạo danh mục phù hợp.
\end{itemize}
\subsection{Hạn chế của hệ thống}
\begin{itemize}
	\item Chưa có tính đa người dùng.
	\item AI còn đơn giản, các phân tích còn cơ bản, chưa cá nhân hóa sâu hoặc học từ dữ liệu lâu dài.
	\item Chưa có ứng dụng di động.
	\item Chưa hỗ trợ đa tiền tệ hoặc nhập liệu tự động từ ngân hàng.
	\item Tính năng thống kê còn đơn giản.
\end{itemize}
\section{Định hướng tương lai}
\begin{itemize}
	\item Nâng cấp tính năng AI.
	\item Phát triển ứng dụng di động.
	\item Tích hợp nhập thông tin tự động từ ngân hàng hoặc ví điện tử.
	\item Hỗ trợ đa tiền tệ và quy đổi ngoại tệ.
	\item Phát triển thêm tính năng thống kê.
\end{itemize}
