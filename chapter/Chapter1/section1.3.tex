\section{Giới thiệu hệ thống quản lý tài chính Finhub}
\begin{itemize}
    \item Một hệ thống quản lý tài chính cá nhân hiện đại, nhằm hỗ trợ theo dõi, quản lý và tối ưu hóa hoạt động tài chính hàng ngày hiệu quả và tiện lợi.
    
    \item Giao diện thân thiện, dễ sử dụng trên nền tảng web, giúp người dùng dễ dàng nhập các khoản thu, chi, lập kế hoạch và theo dõi tình hình tài chính.

    \item Tích hợp AI cải thiện trải nghiệm người dùng, hỗ trợ trong việc quản lý thu chi triệt để hơn.  

    \item Chuẩn bảo mật hiện đại, đảm bảo thông tin người dùng được bảo vệ.\\
\end{itemize}
