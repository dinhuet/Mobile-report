\section{Mục tiêu đề tài}
\begin{itemize}
    \item Xây dựng hệ thống giúp người dùng quản lý các khoản thu chi cá nhân.
    
    \item Cung cấp giao diện thân thiện, dễ sử dụng cho người dùng.
    
    \item Hỗ trợ thống kê dữ liệu tài chính.
    
    \item Cung cấp dịch vụ trò chuyện với Chatbot, hỗ trợ quản lí chi tiêu và đưa ra gợi ý tài chính
\end{itemize}
