\section{Thu thập và phân tích đặc tả yêu cầu}
\subsection{Yêu cầu người sử dụng}
\begin{itemize}
	\item Có thể đăng nhập vào hệ thống an toàn.
	\item Có thể thêm một giao dịch
	\item Có thể chỉnh sửa hoặc xóa các giao dịch đã tạo.
	\item Có thể tạo, sửa, xóa các danh mục
	\item Có thể xem báo cáo chi tiêu theo khoảng thời gian tùy chỉnh.
	\item Giao diện thân thiện 
	\item Hệ thống tích hợp AI
\end{itemize}
\subsection{Yêu cầu hệ thống}
\subsubsection{Yêu cầu chức năng}
\begin{itemize}
	\item Người dùng có thể đăng nhập bằng email và mật khẩu
	\item Bảo mật thông tin người dùng 
	\item Thêm giao dịch mới
	\item Sửa và xóa giao dịch đã lưu
	\item Thêm danh mục mới
	\item Sửa và xóa danh mục đã tạo
	\item Báo cáo tài chính theo khoảng thời gian tự chọn, hiển thị bằng biểu đồ tròn
\end{itemize}
\subsubsection{Yêu cầu phi chức năng}
\begin{itemize}
	\item Khả năng sử dụng: Giao diện đơn giản, dễ thao tác trên máy tính và thiết bị di động
	\item Hiệu suất: Hệ thống xử lý yêu cầu trong thời gian tối đa 2 giây
	\item Bảo mật: Dữ liệu được bảo mật
\end{itemize}
